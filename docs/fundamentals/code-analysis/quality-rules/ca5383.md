    public void ExampleMethod(string key, string value)
    {
        var cookieOptions = new CookieOptions();
        cookieOptions.Secure = false;
        Random r = new Random();

        if (r.Next(6) == 4)

            cookieOptions.Secure = true;
        }

        responseCookies.Append(key, value, cookieOptions);
